Events

Code example:

$(function(){
  // Single event binding:
  $.on($('#button1'), 'click', function(event) {
    alert(this.textContent);
  });
  // Single event binding:
  $.on($('#button2'), 'click', function(event) {
    alert(this.textContent);
  });
  // Delegate binding:
  $.on($('#list'), 'li', 'click', function(event) {
    alert(this.textContent);
  });
  // Trigger event on another element:
  $.on($('#button3'), 'click', function(event) {
    $.trigger($('#button1'), 'click');
  });
});
