The $$ selector always returns an array of matches, even if only one.

Code example:

$(function(){
  var li = $$('li');
  var div = $$('div');
  var button = $$('.button');
  var result = $$('#result span');
  li.forEach(function(ctx) {
    result[0].textContent += (ctx.textContent + ", ");
  });
  div.forEach(function(ctx) {
    result[1].textContent += (ctx.textContent + ", ");
  });
  button.forEach(function(ctx) {
    result[2].textContent += (ctx.textContent + ", ");
  });
});
