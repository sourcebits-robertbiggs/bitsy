The $ selector always returns the first match.

Code example:

$(function(){
  var li = $('li');
  var div = $('div');
  var button = $('.button');
  var result = $('#result');
  result.textContent += li.textContent + ", ";
  result.textContent += div.textContent + ", ";
  result.textContent += button.textContent;
});
